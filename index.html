    <script>

        function searchBrokers() {

            const input = document.getElementById('searchBar').value.toUpperCase();

            const table = document.getElementById('brokerTable');

            const tr = table.getElementsByTagName('tr');


            for (let i = 0; i < tr.length; i++) {

                const tdName = tr[i].getElementsByTagName('td')[0];

                const tdDOT = tr[i].getElementsByTagName('td')[1];

                const tdMC = tr[i].getElementsByTagName('td')[2];


                if (tdName || tdDOT || tdMC) {

                    const txtValueName = tdName.textContent || tdName.innerText;

                    const txtValueDOT = tdDOT.textContent || tdDOT.innerText;

                    const txtValueMC = tdMC.textContent || tdMC.innerText;


                    if (txtValueName.toUpperCase().indexOf(input) > -1 || 

                        txtValueDOT.toUpperCase().indexOf(input) > -1 || 

                        txtValueMC.toUpperCase().indexOf(input) > -1) {

                        tr[i].style.display = "";

                    } else {

                        tr[i].style.display = "none";

                    }

                }

            }

        }

    </script>
